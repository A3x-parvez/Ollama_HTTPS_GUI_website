public class RecursionExample {
    public static void main(String[] args) {
        int n = 5; // test with any positive integer value for 'n' here. Change this value as per your requirement or try user input to see how it works in real-time.  
        
        System.out.println("Factorial of " + n + " is: " + factorial(n));

    public static int factorial(int n) {
        if (n == 0 || n == 1) { // base case where the recursion stops to avoid infinite loop. Factorials for zero and one are always equal to themselves, i.e., both of them being considered as '1'.
            return 1;
        } else {
            return (n * factorial(n - 1)); // recursive call with decremented value until it reaches the base case defined above. This is where multiplication comes into play for calculation, i.e., each subsequent function will multiply its input 'n' by result of previous function which in turn multiplies itself and so on till the point we reach to factorial(1) or 0
        }
    }}
